// 🚀 VideoSlide Notes 核心逻辑 Console 验证脚本
(async function testBiliNote() {
    console.log("=============================================");
    console.log("🎬 正在启动 VideoSlide Notes 核心链路验证...");
    
    // 1. 获取 BVID 和 CID (Console 环境下可以直接拿页面全局变量)
    const state = window.__INITIAL_STATE__;
    if (!state || !state.bvid || !state.cid) {
        console.error("❌ 无法获取 bvid 或 cid，请确保当前在 B站标准视频播放页！");
        return;
    }
    const { bvid, cid } = state;
    console.log(`✅ [1/4] 成功获取视频信息: BVID=${bvid}, CID=${cid}`);

    // 2. 获取字幕
    let subtitles = [];
    try {
        console.log("⏳ 正在请求字幕列表 API...");
        // 带着当前的 B站登录态请求
        const res = await fetch(`https://api.bilibili.com/x/player/v2?cid=${cid}&bvid=${bvid}`, {credentials: 'include'});
        const data = await res.json();

        const subList = data.data?.subtitle?.subtitles;
        if (subList && subList.length > 0) {
            const subUrl = subList[0].subtitle_url;
            const finalUrl = subUrl.startsWith('http') ? subUrl : `https:${subUrl}`;
            console.log(`⏳ 正在下载字幕文件...`);
            const subRes = await fetch(finalUrl);
            const subData = await subRes.json();
            subtitles = subData.body;
            console.log(`✅ [2/4] 字幕加载成功！共提取 ${subtitles.length} 条字幕句子。`);
        } else {
            console.warn("⚠️ [2/4] 该视频没有提供 CC 字幕/AI 字幕，请换个视频测试，或仅依赖截图。");
        }
    } catch (e) {
        console.error("❌ 获取字幕失败: ", e);
    }

    // 3. 截取画面与时间
    const video = document.querySelector('video');
    if (!video) {
        console.error("❌ 找不到 <video> 元素！");
        return;
    }
    const currentTime = video.currentTime;
    console.log(`✅ [3/4] 当前播放时间: ${currentTime.toFixed(2)} 秒`);

    const canvas = document.createElement('canvas');
    canvas.width = video.videoWidth;
    canvas.height = video.videoHeight;
    const ctx = canvas.getContext('2d');
    ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
    
    let base64 = "";
    try {
        base64 = canvas.toDataURL('image/jpeg', 0.5); // 压缩质量0.5
        console.log(`✅ [4/4] 画面截取成功！Base64 图片长度: ${base64.length}`);
    } catch(e) {
        console.error("❌ 画面截取失败 (可能遇到跨域或安全限制): ", e);
    }

    // 4. 提取上下文 (这里设为前 30秒 到 后 10秒)
    let contextText = "无字幕上下文";
    if (subtitles.length > 0) {
        const beforeSec = 30; 
        const afterSec = 10;  
        const start = currentTime - beforeSec;
        const end = currentTime + afterSec;
        const chunks = subtitles.filter(s => s.to > start && s.from < end);
        contextText = chunks.map(s => s.content).join(" ");
    }

    // 5. 打印最终组装发给 LLM 的 Payload 模拟
    console.log("\n🎯 【模拟组装完毕！即将发送给 LLM 的 Payload 如下】");
    console.log({
        videoTitle: document.title,
        timestamp: `${Math.floor(currentTime/60)}分${Math.floor(currentTime%60)}秒`,
        contextSubtitles: contextText || "（无）",
        imagePreview: "（Base64太长，已在下方通过 CSS 渲染出缩略图 ↓）"
    });
    console.log("=============================================");

    // 🎆 彩蛋：在 Console 里直接把图片预览出来！
    if (base64) {
        console.log("%c ", `
            font-size: 1px;
            padding: 120px 200px;
            background-image: url(${base64});
            background-size: contain;
            background-repeat: no-repeat;
            background-position: center;
            border: 2px solid #fb7299;
            border-radius: 8px;
        `);
    }
})();
